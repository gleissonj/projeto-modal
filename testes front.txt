import { LightningElement, api } from 'lwc';

export default class TabelaComSelecao extends LightningElement {
    @api processedData;  // Recebe os dados da tabela

    // Computa se há dados para mostrar
    get hasDataToShow() {
        // Verifica se existe pelo menos uma linha com dados em mobileOptions ou emailOptions
        return this.processedData.some(row => row.mobileOptions.length > 0 || row.emailOptions.length > 0);
    }

    handleSelectChange(event) {
        // Lógica para lidar com a alteração de seleção
        const selectedValue = event.target.value;
        const rowId = event.target.dataset.id;

        // Encontre a linha correspondente e atualize o valor selecionado
        const row = this.processedData.find(item => item.id === rowId);
        if (event.target.name === 'selectedPhone') {
            row.selectedPhone = selectedValue;
        } else if (event.target.name === 'selectedEmail') {
            row.selectedEmail = selectedValue;
        }
    }

    handleConfirm() {
        // Lógica para o botão de confirmação
        console.log('Dados confirmados:', this.processedData);
    }
}
