describe('getValueNameOperation', () => {
  let component: YourComponent;

  beforeEach(() => {
    component = new YourComponent(); // ajuste conforme seu componente real
  });

  it('deve retornar a parte antes do ponto quando houver ponto no nome', () => {
    const result = component.getValueNameOperation('documento.pdf');
    expect(result).toBe('documento');
  });

  it('deve retornar undefined quando não houver ponto no nome', () => {
    const result = component.getValueNameOperation('documento');
    expect(result).toBeUndefined();
  });

  it('deve lidar com valor nulo ou indefinido sem lançar erro', () => {
    const result = component.getValueNameOperation(null);
    expect(result).toBeUndefined();
  });
});
