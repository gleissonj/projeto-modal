import { ComponentFixture, TestBed } from '@angular/core/testing';
import { MyComponent } from './my-component.component'; // Adjust the import path
import { AuthService } from './auth.service'; // Adjust the import path

describe('MyComponent', () => {
  let component: MyComponent;
  let fixture: ComponentFixture<MyComponent>;
  let authService: AuthService;

  beforeEach(() => {
    // Mock the AuthService
    const authServiceMock = {
      logoutRedirect: jest.fn()
    };

    TestBed.configureTestingModule({
      declarations: [MyComponent],
      providers: [
        { provide: AuthService, useValue: authServiceMock },
      ],
    }).compileComponents();

    fixture = TestBed.createComponent(MyComponent);
    component = fixture.componentInstance;
    authService = TestBed.inject(AuthService);

    fixture.detectChanges();
  });

  it('should call logoutRedirect with the correct parameters', () => {
    // Arrange
    const logoutRedirectSpy = jest.spyOn(authService, 'logoutRedirect');
    const expectedRedirectUri = 'http://localhost:4200';

    // Act
    component.logout();

    // Assert
    expect(logoutRedirectSpy).toHaveBeenCalledWith({
      postLogoutRedirectUri: expectedRedirectUri
    });
  });
});
