import { TestBed } from '@angular/core/testing';
import { HttpClientTestingModule, HttpTestingController } from '@angular/common/http/testing';
import { MyService } from './my-service.service'; // Adjust the path to your service

describe('MyService', () => {
  let service: MyService;
  let httpTestingController: HttpTestingController;

  beforeEach(() => {
    TestBed.configureTestingModule({
      imports: [HttpClientTestingModule],
      providers: [MyService],
    });

    service = TestBed.inject(MyService);
    httpTestingController = TestBed.inject(HttpTestingController);
  });

  afterEach(() => {
    httpTestingController.verify(); // Verifies that no unmatched requests remain
  });

  it('should return the expected JSON data from getJson', (done) => {
    const mockData = { key: 'value' }; // Example mock data

    service.getJson().subscribe((data) => {
      expect(data).toEqual(mockData);
      done(); // Call done to indicate the test is complete
    });

    // Expect one GET request to the specified URL
    const req = httpTestingController.expectOne('./assets/exemplo.json');
    expect(req.request.method).toBe('GET');

    // Respond with the mock data
    req.flush(mockData);
  });
});
