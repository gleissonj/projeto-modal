results.forEach((response, index) => {
  const fileName = files[index];

  if (response && response.url) {
    // Se a URL é uma URL direta de um arquivo binário
    fetch(response.url)
      .then((res) => res.blob())  // Converte a resposta em Blob
      .then((blob) => {
        const link = document.createElement('a');
        const url = URL.createObjectURL(blob);  // Cria um URL para o blob
        link.href = url;
        link.download = fileName;
        link.click();
        URL.revokeObjectURL(url);  // Libera o objeto URL depois do uso
      })
      .catch((err) => {
        console.error(`Erro ao baixar o arquivo ${fileName}`, err);
      });
  } else {
    console.error(`URL inválida para o arquivo ${fileName}`);
  }
});
