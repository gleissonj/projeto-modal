import { MyComponent } from './my-component'; // Ajuste o caminho do seu componente

describe('MyComponent', () => {
  let component: MyComponent;

  beforeEach(() => {
    // Inicializa o componente antes de cada teste
    component = new MyComponent();
    
    // Mock do método showMessage para que possamos verificar se foi chamado
    jest.spyOn(component, 'showMessage');
  });

  it('deve chamar showMessage com "Nao contem o arquivo no bucket" e "error"', () => {
    // Chama a função downloadFileMigration
    component.downloadFileMigration();
    
    // Verifica se o showMessage foi chamado com a mensagem e tipo de erro
    expect(component.showMessage).toHaveBeenCalledWith('Nao contem o arquivo no bucket', 'error');
  });
});
