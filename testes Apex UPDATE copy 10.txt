it('deve adicionar arquivos e atualizar hash se nome for diferente', () => {
  const fakeFile = new File(['conteudo'], 'arquivo_teste.txt', { type: 'text/plain' });
  const eventoMock = {
    target: {
      files: [fakeFile]
    }
  };

  component.selectedFiles = [];
  component.hash = 'outro_arquivo.txt'; // diferente do nome do arquivo enviado

  component.escolherArquivo(eventoMock);

  // Verifica se o arquivo foi adicionado ao selectedFiles
  expect(component.selectedFiles.length).toBe(1);
  expect(component.selectedFiles[0].name).toBe('arquivo_teste.txt');

  // Verifica se o hash foi atualizado
  expect(component.hash).toBe('arquivo_teste.txt');
});
