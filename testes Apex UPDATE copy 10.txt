describe('getValueNameOperation', () => {
  let component: any;

  beforeEach(() => {
    component = {
      getValueNameOperation(name: any) {
        let index = name.indexOf('.');
        if (index !== -1) {
          return name.substring(0, index);
        }
      }
    };
  });

  it('deve retornar a parte antes do ponto se existir ponto', () => {
    const result = component.getValueNameOperation('documento.pdf');
    expect(result).toBe('documento');
  });

  it('deve retornar undefined se nÃ£o houver ponto', () => {
    const result = component.getValueNameOperation('documento');
    expect(result).toBeUndefined();
  });
});



it('deve retornar "documento" ao passar uma string com ponto', () => {
  const input = 'documento.pdf';
  expect(input.indexOf('.')).not.toBe(-1); // extra check para debug
  const result = component.getValueNameOperation(input);
  expect(result).toBe('documento');
});
