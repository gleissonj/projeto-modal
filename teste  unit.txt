describe('generateNewPathValues', () => {
    it('should generate correct path values based on currentStatus', () => {
        // Create an instance of your component or class
        const component = new YourComponentOrClass();

        // Call the method with a specific currentStatus
        const currentStatus = 'Finalizado';
        component.generateNewPathValues(currentStatus);

        // Expected pathValues array
        const expectedPathValues = [
            {"attributes": null, "label": 'Nao iniciado', "value": 'Nao iniciado'},
            {"attributes": null, "label": 'Em prospeccao', "value": 'Em prospeccao'},
            {"attributes": null, "label": currentStatus, "value": currentStatus},
        ];

        // Assertions
        expect(component.pathValues).toEqual(expectedPathValues);
    });
});
